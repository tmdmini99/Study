


```js
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Dynamic Script Loading</title>
</head>
<body>
    <div id="content">
        <!-- 페이지 콘텐츠 -->
    </div>

    <script>
        document.addEventListener('DOMContentLoaded', function () {
            const currentUrl = window.location.pathname;

            // 서버에서 경로 패턴을 가져옴
            fetch('/api/routes') // 경로 패턴을 반환하는 API를 호출
                .then(response => response.json())
                .then(patterns => {
                    let scriptSrc;

                    // 각 경로 패턴에 대해 조건문 실행
                    patterns.forEach(pattern => {
                        // 현재 URL이 기본 패턴과 일치하는지 체크
                        if (currentUrl.startsWith(pattern)) {
                            // 기본 패턴인 경우
                            if (currentUrl.split('/').length === pattern.split('/').length) {
                                scriptSrc = '/resources/js/pagination.js'; // 기본 스크립트
                            } else {
                                // 디테일 페이지인 경우
                                scriptSrc = '/resources/js/paginationDetail.js'; // 디테일 스크립트
                            }
                        }
                    });

                    // 결정된 스크립트 로드
                    if (scriptSrc) {
                        const scriptElement = document.createElement('script');
                        scriptElement.src = scriptSrc;
                        document.body.appendChild(scriptElement);
                    } else {
                        console.log("적합한 스크립트가 없습니다.");
                    }
                })
                .catch(error => {
                    console.error('경로 패턴을 가져오는 중 오류 발생:', error);
                });
        });
    </script>
</body>
</html>

```
