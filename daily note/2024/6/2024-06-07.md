$(".grid_wrap").on("dblclick", "ul > li > ul > li a", function() {  
  
        var $link = $(this);  
        var text = $link.text();  
        var input = $("<input>").attr("type", "text").val(text);  
        console.log(input.val());  
        $link.empty().append(input);  
        input.focus();  
        input.on("blur", function() {  
            var newText = $(this).val();  
            $link.text(newText); // 링크의 텍스트를 새로운 값으로 설정합니다.  
        });  
    });  
  
    $(".add").on("click", function() {  
    var li = $("<li>").addClass("ui-sortable-handle");  
    var input = $("<input>").attr("type", "text").css("width", "100px");  
    var link = $("<a>").attr("href", "#").addClass("grid_link").append(input);  
    li.append(link);  
  
    var ul = $(".grid_wrap .depth02_bg:first").find("> ul:nth-child(2) > li:first > ul");  
    ul.append(li);  
  
    input.on("change", function() {  
        var newText = $(this).val();  
        var newLink = $("<a>").attr("href", "#").text(newText);  
        $(this).parent().replaceWith(newLink);  
    });  
});
